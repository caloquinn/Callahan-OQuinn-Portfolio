<body>
<input type ="text" id="screen"> <!-- creation of the screen the input/output will be-->
<table>  <!-- creates a table (grid) that the buttons will exist in-->
    <tr>
        <td> <button onclick="numberPressed('1')">1</button></td> <!-- creating the buttons, displays whats on the button & will call the appropriate functions-->
        <td> <button onclick="numberPressed('2')">2</button></td>
        <td> <button onclick="numberPressed('3')">3</button></td>
        <td> <button onclick="numberPressed('4')">4</button></td>
        <td> <button onclick="numberPressed('5')">5</button></td>
    </tr>
    <tr>
        <td> <button onclick="numberPressed('6')">6</button></td>
        <td> <button onclick="numberPressed('7')">7</button></td>
        <td> <button onclick="numberPressed('8')">8</button></td>
        <td> <button onclick="numberPressed('9')">9</button></td>
        <td> <button onclick="numberPressed('0')">0</button></td>
    </tr>
    <tr>
        <td> <button onclick="operatorPressed('+')">+</button></td>
        <td> <button onclick="operatorPressed('-')">-</button></td>
        <td> <button onclick="operatorPressed('*')">*</button></td>
        <td> <button onclick="operatorPressed('/')">/</button></td>
        <td> <button onclick="equalPressed('=')">=</button></td>  
    </tr>
</table>
<script>
    let value1 = null; // variable for the first variable in the equation
    let value2 = null; //variable for the second number in the equation
    let operator = null; //variable for the operator in the equation
    let inputNumber = ""; //variable that stores a number that is still being inputed

    function numberPressed(num) //functions for the number buttons
    {
        inputNumber += num; // this ensures subsequent numbers clicked will append to the first number
        document.getElementById("screen").value = inputNumber //this id's the screen on the calculator and will display the inputted number there
    }

    function operatorPressed(op) //function for operators being pressed, this handles "closing" the input of the first number
    {
        if (inputNumber === "") return; // if a number is not pressed before an operator, do nothing
        value1 = parseFloat(inputNumber); //using parseFloat to "translate" the string of inputted numbers into an actual number and stores it as value1
        operator = op; //stores the pressed operator
        inputNumber =""; // "clears" the screen 
        document.getElementById("screen").value = value1 + " " + operator; // adds the operator onto the end of the inputted number that is displayed on the screen
    }
    function equalPressed() // function that occurs when the = sign is pressed, signifying the end of inputs and triggers the calculation 
    {
        if (inputNumber === "" || value1 === null || operator === null) return; // this makes sure everything needed (number-operator-number) is inputted prior to the equal sign
        value2 = parseFloat(inputNumber);
        let result = doMath(value1, operator, value2); // this calls the domath function to do the calculation and return the result 
        document.getElementById("screen").value = result; // displays the result of the calculations
    }
    function doMath(value1, operator, value2) // the function that will actually do the equation
    {
        switch (operator) //using "switch" to check for which operator is clicked
        {
            case '+': return value1 + value2; // case is used like "if" with "switch" - if the operator in case matches, it will perform the said operation with the right operator
            case '-': return value1 - value2;
            case '*': return value1 * value2;
            case '/': return value2 !==0 ? value1 / value2: "error"; // this checks if the second numebr is 0, if it is not, it will do the division
            default: return "error"; // returns an error if something doesnt match, should not occur though becauase buttons are hard coded 
        }
    }
</script>
